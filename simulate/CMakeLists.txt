cmake_minimum_required(VERSION 3.16)
project(unitree_mujoco)

enable_language(C)
enable_language(CXX)
set(CMAKE_CXX_STANDARD 17)

find_package(mujoco REQUIRED)
find_package(OsqpEigen REQUIRED)
find_package(Eigen3)

include_directories(/usr/local/include/ddscxx 
                    /usr/local/include/iceoryx/v2.0.2 
                    ${EIGEN3_INCLUDE_DIR}
                    /home/jang/unitree_mujoco/simulate/src/controller)
link_libraries(unitree_sdk2 ddsc ddscxx rt pthread OsqpEigen::OsqpEigen)

FILE (GLOB SIM_SRC
        src/joystick/joystick.cc
        src/mujoco/*.cc
        src/unitree_sdk2_bridge/*.cc
        src/controller/*.cpp)

set(SIM_DEPENDENCIES 
    pthread
    mujoco::mujoco
    glfw
    yaml-cpp)


add_executable(unitree_mujoco ${SIM_SRC} src/main.cc)
target_link_libraries(unitree_mujoco  ${SIM_DEPENDENCIES})

add_executable(test test/test_unitree_sdk2.cpp)
add_executable(jstest src/joystick/jstest.cc src/joystick/joystick.cc)

SET(CMAKE_BUILD_TYPE Release)
